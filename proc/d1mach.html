pure function d1mach(i) result(d)

    use iso_fortran_env, only: rp => real64

    implicit none 

    integer,intent(in) :: i
    real(rp) :: d

    real(rp),dimension(5),parameter :: constants = &
        [tiny(1.0_rp), &
         huge(1.0_rp), &
         real(radix(1.0_rp),rp)**(-digits(1.0_rp)), &
         epsilon(1.0_rp), &
         log10(real(radix(1.0_rp),rp))]

    d = constants(i)

end function d1mach
