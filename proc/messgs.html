    function messgs() result(iunit)

    integer :: iunit

    integer :: istat !! iostat code

    open (newunit=iunit, &
          status='SCRATCH', &
          access='SEQUENTIAL', &
          form='UNFORMATTED', &
          iostat=istat)

    if (istat /= 0) error stop 'Could not assign scratch unit in MESS.'

      end function messgs
